{

	def string buff;

	for i in 0..length step ++i
	{

		{
			if(numbers>3) => return false;
			buff[numbers] = address[i];
			++numbers;
		}
		elif(address[i] == '.')
		{
			if(ToNumber(buff) > 255 or numbers == 0) => return false ;
			numbers = 0;
			++octets;

		}
		else => return false;
	} 
	if(ToNumber(buff) > 255 or octets != 3 or numbers == 0) => return false;
	return true;
}
